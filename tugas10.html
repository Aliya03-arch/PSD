IDENTIFIKASI SUARA ‘BUKA TUTUP’ MENGGUNAKAN FEATURE STATISTIK TIME SERIES#

Berikut adalah sumber datasetnya nya : https://www.kaggle.com/datasets/muhammadridhoisdi/audio-recognition-open-and-close.

Ekstraksi Fitur dari Audio#

def extract_features(file_path):
    x, sr = librosa.load(file_path, sr=None)

    features = {
        "mean": np.mean(x),
        "std": np.std(x),
        "max": np.max(x),
        "min": np.min(x),
        "rms": np.sqrt(np.mean(x**2)),
        "zcr": np.mean(librosa.feature.zero_crossing_rate(x)),
        "skew": pd.Series(x).skew(),
        "kurt": pd.Series(x).kurt()
    }
    return features

# --- Normalisasi data sebelum training ---
# Pastikan X_train/X_test atau X_train_s/X_test_s sudah tersedia dari cell sebelumnya.
# Jika X_train/X_test ada, lakukan scaling di sini; jika sudah tersedia X_train_s/X_test_s, pakai yang ada.
if 'X_train' in globals() and 'X_test' in globals():
    scaler = MinMaxScaler()
    X_train_s = scaler.fit_transform(X_train)
    X_test_s = scaler.transform(X_test)
elif 'X_train_s' in globals() and 'X_test_s' in globals():
    # sudah diskalakan di cell lain, gunakan langsung
    X_train_s = X_train_s
    X_test_s = X_test_s

for name, model in models.items():
    # Train and predict using the scaled features
    model.fit(X_train_s, y_train)
    y_pred = model.predict(X_test_s)

    # Prediksi probabilitas untuk mendeteksi "unknown"
    if hasattr(model, "predict_proba"):

        max_proba = np.max(y_proba, axis=1)
    else:
        # fallback jika model tidak mendukung predict_proba
        max_proba = np.ones(len(y_pred))

    # Threshold (bisa disesuaikan)
    THRESHOLD = 0.6  # jika confidence < 0.6 dianggap unknown

    pred_labels = []
    for p, prob in zip(max_proba, y_pred):
        if p < THRESHOLD:
            pred_labels.append("unknown")
        else:
            pred_labels.append(prob)  # y_pred already contains string labels

    df_eval = pd.DataFrame({
        "true": y_test,
        "predicted": pred_labels,
        "confidence": max_proba
    })

    print(df_eval.head(20))

    acc = accuracy_score(y_test, y_pred)
    results[name] = acc
    print(f"\n=== {name} ===")
    print("Accuracy:", acc)
    print(classification_report(y_test, y_pred))

def extract_features_file(path, sr=16000, n_mfcc=20):
    y, sr = librosa.load(path, sr=sr, mono=True)
    # safety: jika audio terlalu pendek, pad dengan zeros
    if y.shape[0] < 1024:
        y = np.pad(y, (0, 1024 - y.shape[0]))

    # MFCC
    mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=n_mfcc)
    mfcc_mean = np.mean(mfcc, axis=1)
    mfcc_std = np.std(mfcc, axis=1)

    # Chroma
    chroma = librosa.feature.chroma_stft(y=y, sr=sr)
    chroma_mean = np.mean(chroma, axis=1)

    # Spectral features
    spec_centroid = np.mean(librosa.feature.spectral_centroid(y=y, sr=sr))
    spec_bandwidth = np.mean(librosa.feature.spectral_bandwidth(y=y, sr=sr))
    spec_contrast = np.mean(librosa.feature.spectral_contrast(y=y, sr=sr))
    spec_rolloff = np.mean(librosa.feature.spectral_rolloff(y=y, sr=sr))
    zcr = np.mean(librosa.feature.zero_crossing_rate(y))
    rms = np.mean(librosa.feature.rms(y=y))

    # combine into single feature vector
    feature_vector = np.hstack([
        mfcc_mean, mfcc_std,
        chroma_mean,
        spec_centroid, spec_bandwidth, spec_contrast, spec_rolloff,
        zcr, rms
    ])

    return feature_vector

    rows = []
    # Expect folder structure: dataset/<speaker>/<files.wav>
    for speaker_folder in sorted(os.listdir(DATASET_DIR)):
        speaker_path = os.path.join(DATASET_DIR, speaker_folder)
        if not os.path.isdir(speaker_path):
            continue
        # iterate wav files
        for wav_path in glob.glob(os.path.join(speaker_path, "*.wav")):
            try:
                feats = extract_features_file(wav_path)
            except Exception as e:
                print("Error reading", wav_path, e)
                continue
            # infer label from filename or parent folder naming convention
            # We assume filenames like open_1.wav, close_1.wav OR folder names encode speaker only
            fname = os.path.basename(wav_path).lower()
            if "open" in fname:
                label = "open"
            elif "close" in fname or "closed" in fname:
                label = "close"
            else:
                # fallback: try parent folder name contains label (e.g., A_open)
                if "_" in speaker_folder:
                    parts = speaker_folder.split("_")
                    if parts[-1] in ("open", "close"):
                        # If folder named speakerA_open etc.
                        label = parts[-1]
                    else:
                        label = "unknown"
                else:
                    label = "unknown"

            rows.append({
                "path": wav_path,
                "speaker": speaker_folder,
                "label": label,
                **{f"f{i}": float(feats[i]) for i in range(len(feats))}
            })

    df = pd.DataFrame(rows)
    # drop unknown label rows (if any)
    df = df[df["label"].isin(["open", "close"])].reset_index(drop=True)
    # save
    df.to_csv(FEATURES_CSV, index=False)
    print("Saved features to", FEATURES_CSV)

# feature columns are f0..fN
feature_cols = [c for c in df.columns if c.startswith("f")]

le_speaker = LabelEncoder()

    X, y_combined, test_size=0.5, random_state=42, stratify=y_combined

    results[name] = {"model": model, "accuracy": acc}

def extract_features_file(path, sr=16000, n_mfcc=20):
    y, sr = librosa.load(path, sr=sr, mono=True)
    if y.shape[0] < 1024:
        y = np.pad(y, (0, 1024 - y.shape[0]))

    mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=n_mfcc)
    mfcc_mean = np.mean(mfcc, axis=1)
    mfcc_std = np.std(mfcc, axis=1)

    chroma = librosa.feature.chroma_stft(y=y, sr=sr)
    chroma_mean = np.mean(chroma, axis=1)

    spec_centroid = np.mean(librosa.feature.spectral_centroid(y=y, sr=sr))
    spec_bandwidth = np.mean(librosa.feature.spectral_bandwidth(y=y, sr=sr))
    spec_contrast = np.mean(librosa.feature.spectral_contrast(y=y, sr=sr))
    spec_rolloff = np.mean(librosa.feature.spectral_rolloff(y=y, sr=sr))
    zcr = np.mean(librosa.feature.zero_crossing_rate(y))
    rms = np.mean(librosa.feature.rms(y=y))

    return np.hstack([
        mfcc_mean, mfcc_std, chroma_mean,
        spec_centroid, spec_bandwidth, spec_contrast, spec_rolloff,
        zcr, rms
    ]).reshape(1, -1)
